cmake_minimum_required(VERSION 3.22)
project(whisper_android_jni)

# ── Whisper.cpp build options ─────────────────────────────────────────────────
set(WHISPER_BUILD_TESTS    OFF CACHE BOOL "" FORCE)
set(WHISPER_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
set(BUILD_SHARED_LIBS      OFF CACHE BOOL "" FORCE)

# Disable GPU back-ends not available on Android
set(GGML_METAL  OFF CACHE BOOL "" FORCE)
set(GGML_CUDA   OFF CACHE BOOL "" FORCE)
set(GGML_VULKAN OFF CACHE BOOL "" FORCE)
set(GGML_OPENCL OFF CACHE BOOL "" FORCE)

# ── Include whisper.cpp as a sub-project ──────────────────────────────────────
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/whisper.cpp)

# ── JNI bridge library ───────────────────────────────────────────────────────
find_library(log-lib log)

add_library(whisper_jni SHARED whisper_jni.cpp)

target_link_libraries(whisper_jni
    PRIVATE
        whisper
        ${log-lib}
)

target_include_directories(whisper_jni
    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/whisper.cpp/include
)

target_link_options(whisper_jni PRIVATE "-Wl,-z,max-page-size=16384")
